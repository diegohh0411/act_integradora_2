# Estructura de control: if-elif-else
def control_flujo(valor):
    if valor > 10:
        print("Mayor que 10")
    elif valor > 5:
        print("Entre 5 y 10")
    else:
        print("Menor o igual a 5")

# Bucles: for y while
def bucles_ejemplo():
    # Bucle for
    for i in range(10):
        if i == 5:
            continue  # Salta a la siguiente iteración
        if i == 8:
            break     # Sale del bucle
        print(i)
    
    # Bucle while
    contador = 0
    while contador &lt; 5:
        print(f&quot;Contador: {contador}&quot;)
        contador += 1

# Manejo de excepciones: try-except-finally
def manejo_excepciones():
    try:
        resultado = 10 / 0
    except ZeroDivisionError as e:
        print(f&quot;Error: {e}&quot;)
    except Exception as e:
        print(f&quot;Error general: {e}&quot;)
    else:
        print("No hubo errores")
    finally:
        print("Siempre se ejecuta")

# Generadores y yield
def generador():
    for i in range(5):

# Decoradores
def decorador(func):
    def wrapper(*args, **kwargs):
        print(&quot;Antes de la función&quot;)
        resultado = func(*args, **kwargs)
        print(&quot;Después de la función&quot;)
        return resultado
    return wrapper

@decorador
def funcion_decorada():
    print("Función decorada")

# Variables globales y nonlocal
variable_global = &quot;global&quot;

def funcion_externa():
    variable_local = &quot;local&quot;
    
    def funcion_interna():
        nonlocal variable_local
        global variable_global
        variable_local = &quot;modificada&quot;
        variable_global = &quot;modificada globalmente&quot;
    
    funcion_interna()
    return variable_local
